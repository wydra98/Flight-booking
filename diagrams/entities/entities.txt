@startuml
class Airport {
  __ attributes __
  - id: Long
  - name: String
  - city: String
  - country: String
  - longitude: double
  - latitude: double
  - timezone: int
}

class Airline {
  __ attributes __
  - id: Long
  - name: String
  - country: String
}

class Connection {
  __ attributes __
  - id: Long
  - srcAirport: Airport
  - dstAirport: Airport
}

class Flight {
  __ attributes __
  - id: Long
  - connection: Connection
  - airline: Airline
  - numberSeats: int
  - availableSeats: int
  - price: double
  - times: Times
}

class Times {
  __ attributes __
  - departureDate: LocalDate
  - departureTime: LocalTime
  - flightTime: LocalTime
}

class Ticket {
  __ attributes __
  - id: Long
  - passenger: Passenger
  - flight: Flight
  - seatNumber: Integer
  - price: double
}

class Passenger {
  __ attributes __
  -id: Long
  -pesel: String
  -firstname: String
  -surname: String
  -dateOfBirth: LocalDate
  -phoneNumber: String
  -email: String
}

class Trip {
  __ attributes __
  - id: Long
  - code: String
  - tickets: List<Ticket>
  - departureDate: LocalDate
  - departureTime: LocalTime
  - arrivalDate: LocalDate
  - arrivalTime: LocalTime
  - purchaseDate: LocalDate
  - purchaseTime: LocalTime
  - price: Price
}

Connection o-- Airline
Connection o-- Airport
Times --* Flight
Flight o-- Connection
Ticket o-- Passenger
Ticket o-- Flight
Trip *-- Ticket
@enduml
